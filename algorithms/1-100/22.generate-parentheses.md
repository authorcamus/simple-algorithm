给出 n 代表生成括号的对数，请你写出一个函数，使其能够生成所有可能的并且有效的括号组合。

```cpp
例如，给出 n = 3，生成结果为：

[
  "((()))",
  "(()())",
  "(())()",
  "()(())",
  "()()()"
]
```

---

这个题解法很多

## 暴力回溯

```javascript
var generateParenthesis = function(n) {
    let ans = []
    function dfs(path, l, r){
        if (l == n && r == n) {
            ans.push(path.join(''))
            return
        }
        if (l > r) {
            dfs([...path, ')'], l, r+1)
            if (l < n) dfs([...path, '('], l+1, r)
        }else if (l == r) {
            if (l < n) dfs([...path, '('], l+1, r)
        }
    }
    dfs([], 0, 0)
    return ans
};
```

## 深搜

```cpp
class Solution {
public:
    vector<string> ans;
    vector<string> generateParenthesis(int n) {
        dfs("(", 1, 0, n);
        return ans;
    }
    void dfs(string s, int l, int r, int n){
        if (l > n) return;
        if (r == n) {
            ans.push_back(s);
            return ;
        }
        if (l > r) {
            dfs(s + "(", l+1, r, n);
            dfs(s + ")", l , r+1, n);
        } else {
            dfs(s + "(", l+1, r, n);
        }
    }
};
```
