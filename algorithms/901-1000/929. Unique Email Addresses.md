class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> st;
        for (auto s: emails){
            int at = s.find('@');
            string name = "";
            for (int i = 0; i< at; i++){
                if (s[i] == '+') break;
                else if (s[i] !='.') name += s[i];
            }
            string domain = s.substr(at+1);
            st.insert(name + "@" + domain);
        }
        return st.size();
    }
};
